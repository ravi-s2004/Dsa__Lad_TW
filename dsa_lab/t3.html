        #include<stdio.h>
        struct node{
        int data;
        struct node *left,*right;
         };
         struct node *createnode(int e)
         {
             struct node *x;
             x=(struct node*)malloc(sizeof(struct node));
             x->data=e;
             x->right=NULL;
             x->left=NULL;
        
             return x;
         }
         struct node* insertFront(struct node *head,int e)
         {
             struct node *q;
             q=createnode(e);
             if(head==NULL)
             {
                 printf("list empty!\n adding element\n");
                 head=q;
                 display(head);
             }
             q->right=head;
             head->left=q;
             head=q;
         }
         void insertRear(struct node *head,int e)
         {
                struct node *q,*p;
                p=createnode(e);
                if(head==NULL)
                {
                    printf("list empty!\nadding element\n");
                    head=p;
                    return head;
                }
                q=head;
                while(q->right!=NULL)
                {
                    q=q->right;
                }
                p->left=q;
                q->right=p;
         }
         void insertLeft(struct node *head,int n,int e)
         {
             struct node *p,*q;
             p=createnode(e);
             q=head;
             while(q->data!=n)
             {
                 q=q->right;
             }
             p->left=q->left;
             p->right=q;
             q->left->right=p;
             q->left=p;
         }
