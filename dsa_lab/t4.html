        void insert()
        {
            int item;
            if (rear == MAX - 1)
            printf("Queue Overflow \n");
            else
            {
                if (front == - 1)
                    front = 0;
                printf("Inset the element in queue : ");
                scanf("%d", &item);
                queue[++rear] = item;
            }
        } 
         
        void delete()
        {
            if (front == - 1 || front > rear)
            {
                printf("Queue Underflow \n");
                return ;
            }
            else
            
                printf("Element deleted from queue is : %d\n", queue[front++]);
            
        } 
         
        void display()
        {
            int i;
            if (front == - 1)
                printf("Queue is empty \n");
            else
            {
                printf("Queue is : \n");
                for (i = front; i <= rear; i++)
                    printf("%d \n", queue[i]);
            }
