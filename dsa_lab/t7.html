            int Partition( int low , int high)
            {
                int i,j;
                int pivot=A[low];
            
                i=low;
                j=high+1;
            
                while(i<j)
                {
                    do
                    {
                        ++i;
                    }while(A[i]<=pivot);
            
                    do
                    {
                        --j;
                    }while(A[j]>pivot);
            
                    if(i<j)
                    {
                       swap(&A[i],&A[j]);
                       /*t=A[i];
                       A[i]=A[j];
                       A[j]=t;*/
                    }
                }
            
                swap(&A[low],&A[j]);
            
                /*t=A[low];
                A[low]=A[j];
                A[j]=t;*/
            
                return j;
            }
            
            void Quicksort( int low , int high)
            {
                int j;
            
                if(low<high)
                {
                    j=Partition(low,high);
                    Quicksort(low , j-1);
                    Quicksort(j+1 , high);
                }
            }
            
            void swap(int  *a , int   *b)
            {
                int t;
                t= *a;
                *a=*b;
                *b=t;
