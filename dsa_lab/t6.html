    int Binary_Search(int A[],int k,int l,int h)    // Binary Search code
    {
        int m;
        while(l<=h)
        {
            m=(l+h)/2;
            if(A[m]==k)
                return 1; 
            else
            if(k>A[m])
             l=m+1;
            else
             h=m-1;
        }
        return 0;
    
    }
